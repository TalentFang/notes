# 类与对象

## 基本结构

```python
class MyClass:
    """类的文档字符串"""

    # 类变量 (所有实例共享)
    class_attr = "shared"

    def __init__(self, name, value=0):
        """构造函数"""
        # 实例变量
        self.name = name
        self._value = value  # 受保护 (惯例)
        self.__private = "private"  # 名称重整 (name mangling)

    # 实例方法
    def method(self):
        return f"{self.name} = {self._value}"

    # 静态方法 (不需要 self)
    @staticmethod
    def static_method():
        return "static"

    # 类方法 (得到 cls)
    @classmethod
    def from_string(cls, s):
        """工厂方法"""
        parts = s.split(",")
        return cls(parts[0], int(parts[1]))
```

## 继承

```python
class Animal:
    def __init__(self, name):
        self.name = name

    def speak(self):
        raise NotImplementedError

class Dog(Animal):
    def __init__(self, name, breed="Unknown"):
        super().__init__(name)  # 调用父类构造
        self.breed = breed

    def speak(self):
        return f"{self.name} says Woof!"

# 多继承 (注意 MRO)
class Hybrid(Dog, Animal):
    pass
```

## 特殊方法 (魔术方法)

```python
class Vector:
    def __init__(self, x, y):
        self.x = x
        self.y = y

    def __repr__(self):  # 字符串表示
        return f"Vector({self.x}, {self.y})"

    def __add__(self, other):  # + 运算符
        return Vector(self.x + other.x, self.y + other.y)

    def __len__(self):  # len()
        return 2

    def __getitem__(self, idx):  # 索引访问
        return [self.x, self.y][idx]

    def __call__(self):  # 可调用对象
        return self.x + self.y
```

## 属性与 setter

```python
class Person:
    def __init__(self, name):
        self._name = name

    @property
    def name(self):
        """读取属性"""
        return self._name

    @name.setter
    def name(self, value):
        """写入属性"""
        if not value:
            raise ValueError("Name cannot be empty")
        self._name = value

# 使用
p = Person("Alice")
print(p.name)   # 调用 getter
p.name = "Bob"  # 调用 setter
```

## 抽象类

```python
from abc import ABC, abstractmethod

class Shape(ABC):
    @abstractmethod
    def area(self):
        pass

    @property
    @abstractmethod
    def name(self):
        pass

class Circle(Shape):
    def __init__(self, radius):
        self.radius = radius

    def area(self):
        return 3.14 * self.radius ** 2

    @property
    def name(self):
        return "Circle"
```

## 与 Java 对比

| Python | Java |
|--------|------|
| `class MyClass:` | `class MyClass { }` |
| `self` | `this` (隐式) |
| `@property` | `getter/setter` |
| `@staticmethod` | `static` |
| `@classmethod` | 无直接对应 |
| 无 `public/private` 关键字 | `public/private/protected` |
| 多继承 | 单继承 + 接口 |
