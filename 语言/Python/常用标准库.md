# 常用标准库

## os - 操作系统

```python
import os

# 路径操作
os.path.join("dir", "file.txt")      # 路径拼接
os.path.dirname("/a/b/c.txt")         # 目录名
os.path.basename("/a/b/c.txt")       # 文件名
os.path.exists("file.txt")           # 存在?
os.path.isfile("file.txt")           # 是文件?
os.path.isdir("dir")                  # 是目录?

# 环境变量
os.environ.get("HOME")                # 获取环境变量
os.environ["MY_VAR"] = "value"        # 设置环境变量

# 目录操作
os.mkdir("new_dir")                   # 创建目录
os.makedirs("a/b/c")                  # 递归创建
os.rmdir("dir")                       # 删除空目录
os.remove("file.txt")                 # 删除文件

# 执行命令
os.system("ls -la")                   # 执行 shell 命令
os.popen("ls").read()                 # 获取输出

# 文件信息
os.stat("file.txt").st_size           # 文件大小
os.stat("file.txt").st_mtime          # 修改时间
```

## json - JSON 处理

```python
import json

# 序列化
data = {"name": "Alice", "age": 30}
json_str = json.dumps(data, indent=2)  # dict -> str

# 反序列化
parsed = json.loads(json_str)           # str -> dict

# 文件操作
with open("data.json", "w") as f:
    json.dump(data, f, indent=2)       # 写入文件

with open("data.json", "r") as f:
    data = json.load(f)                # 读取文件
```

## pathlib - 路径操作 (现代)

```python
from pathlib import Path

p = Path("/home/user/data/file.txt")

# 路径组件
p.parent           # /home/user/data
p.name             # file.txt
p.stem             # file
p.suffix           # .txt

# 操作
p.exists()         # 存在?
p.is_file()        # 是文件?
p.is_dir()         # 是目录?
p.mkdir(parents=True, exist_ok=True)

# 遍历
list(Path("/tmp").glob("*.txt"))       # 匹配文件
list(Path("/tmp").rglob("*.py"))       # 递归匹配
```

## collections - 容器

```python
from collections import defaultdict, Counter, namedtuple

# 默认字典
d = defaultdict(list)
d["key"].append(1)           # 不需预先检查 key

# 计数器
cnt = Counter("abracadabra")
cnt.most_common(3)           # 最高频的3个

# 命名元组
Point = namedtuple("Point", ["x", "y"])
p = Point(10, 20)
```

## logging - 日志

```python
import logging

logging.basicConfig(
    level=logging.INFO,
    format="%(asctime)s %(levelname)s %(message)s",
    handlers=[
        logging.FileHandler("app.log"),
        logging.StreamHandler()
    ]
)

logger = logging.getLogger(__name__)
logger.info("This is info")
logger.error("This is error")
```

## re - 正则表达式

```python
import re

# 匹配
pattern = r"\d{4}-\d{2}-\d{2}"
if re.match(pattern, "2024-01-15"):
    print("Match!")

# 查找
re.findall(r"\d+", "abc123def456")      # ['123', '456']

# 替换
re.sub(r"\d+", "#", "abc123def456")     # abc#def#

# 分割
re.split(r"[,\s]+", "a,b c,d")          # ['a', 'b', 'c', 'd']
```

## functools - 函数式工具

```python
from functools import lru_cache, wraps, partial

# 缓存
@lru_cache(maxsize=128)
def fib(n):
    return n if n < 2 else fib(n-1) + fib(n-2)

# 偏函数
multiply = partial(int, base=2)
multiply("1010")     # 10

# 冻结参数
from functools import partialmethod
```

## itertools - 迭代器

```python
import itertools

list(itertools.chain([1,2], [3,4]))       # [1,2,3,4]
list(itertools.product([1,2], [3,4]))    # 笛卡尔积
list(itertools.combinations([1,2,3], 2)) # 组合
list(itertools.permutations([1,2,3], 2)) # 排列
```
